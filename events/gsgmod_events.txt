	desc = "END_OF_EMPIRE_DESC"
	picture = BAD_WITH_MONARCH_eventPicture
	
	trigger = {
		NOT = { tag = BYZ }
		OR = {
			government = imperial_government
			government = modern_empire
		}
		NOT = { num_of_cities = 30 }		
	}
	mean_time_to_happen = {
		months = 6
	}
	option = {
		name = "END_OF_EMPIRE_A"
		add_stability = -1
		add_prestige = -30
		if = {
			limit = {
				NOT = { is_year = 1500 }
			}
			change_government = feudal_monarchy
		}
		if = {
			limit = {
				is_year = 1500
				NOT = { is_year = 1600 }
			}
			change_government = administrative_monarchy
		}
		if = {
			limit = {
				is_year = 1600
				NOT = { is_year = 1700 }
			}
			change_government = absolute_monarchy
		}
		if = {
			limit = {
				is_year = 1700
				NOT = { is_year = 1800 }
			}
			change_government = constitutional_monarchy
		}
		if = {
			limit = {
				is_year = 1800
			}
			change_government = enlightened_despotism
		}
	}
}
